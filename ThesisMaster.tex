\chapter*{Traza sobre la cavidad, de la proyección al estado estacionario de la ecuación maestra}

Consideremos el subespacio de eigenvalores $\lambda=0$. La dinámica de tal subespacio es elegido al proyectar la ecuación maestra con $\mathcal{P}_{0}^{\lambda=0}$, es decir

\begin{equation}
\mathcal{P}\rho(t) = \sum_{\lambda\neq0}\mathcal{P}\mathcal{L}_{1}\mathcal{K}_{\lambda}(t)\mathcal{L}_{1}\mathcal{P}\rho(t)
\end{equation}

donde

\begin{equation}
\mathcal{K}_{\lambda}(t)=\sum_{l}K_{l}^{\lambda_{m}}(t)\frac{-\mathcal{P}_0^{\lambda}}{\lambda_{c}+\lambda_{m}+il\omega}
\end{equation}

evidentemente dependen de la forma de la modulación de la frecuencia natural del oscilador mecánico. Y trazando sobre los grados de libertad mecánicos, con la finalidad de obtener la ecuación maestra reducida de la cavidad, obtenemos

\begin{equation}
\label{evolucion}
\dot{\psi}=\sum_{\lambda\neq0}\sum_{l}\frac{K_{l}^{\lambda_{m}}(t)}{\lambda_{c}+\lambda_{m}+il\omega}Tr_{m}\{\mathcal{P}\mathcal{L}_{1}\mathcal{P}_0^{\lambda}\mathcal{L}_{1}\rho_{st}\psi\}
\end{equation}


recuerde que, al igual que en el enfriamiento laser estandar, el proyector

\begin{equation}
\label{estadoseparable}
\mathcal{P}\rho=Tr_{m}\{\mathcal{P}\rho\}\otimes \rho_{st}
\end{equation}

donde $\mathcal{L}_{m}\rho_{st}=0$. Y puesto que en [RefPablo] ha sido calculado el estado estacionario del oscilador mecánico para la misma proyección, podemos saber

\begin{equation}
\rho_{st}=\left(1-\frac{A_{-}}{A_{+}}\right)\left(\frac{A_{-}}{A_{+}}\right)^{b_0^{\dag} b_0}
\end{equation}

Separando los operadores pertenecientes a los grados de libertad mecánicos y de la cavidad, obtenemos

\begin{equation}
\label{separacion}
%\mathcal{P}\mathcal{L}_{1}\mathcal{P}_{0}^{\lambda}\mathcal{L}_{1}\psi\rho_{st}
\hbar^{2}Tr\{\cdots\} = Tr_{1}(\lambda_{m})\mathcal{P}_{c}^{\lambda_{c}=0}\left[F,\mathcal{P}_{c}^{\lambda_{c}}(\psi F)\right]-Tr_{2}( \lambda_{m})\mathcal{P}_{c}^{\lambda_{c}=0}\left[F,\mathcal{P}_{c}^{\lambda_{c}}(F \psi)\right]
\end{equation}

donde se ha utilizado que $Tr\{\mathcal{P}_{m}^{\lambda_{m}=0}A\}=Tr{A}$ para cualquier operador $A$ perteneciente a la parte mecánica. Y por tanto

\begin{equation}
\label{tr1}
Tr_{1}(\lambda_{m})=Tr\{x\mathcal{P}_{m}^{\lambda_{m}}(\rho_{st}x)\}
\end{equation}

\begin{equation}
\label{tr2}
Tr_{2}(\lambda_{m})=Tr\{x\mathcal{P}_{m}^{\lambda_{m}}(x\rho_{st})\}
\end{equation}


Enfocándonos en la parte mecánica, note que las proyecciones $\mathcal{P}_{m}^{\lambda_{m}}$ son sobre el subespacio perteneciente a $\lambda_{m}=il\nu(t)$. Es decir,
cualquier proyección de la forma $\mathcal{P}_{m}^{\lambda_{m}}A=\sum_{n}\hat{\rho}^{(l)}_{n}Tr\{\check{\rho}^{(l)}_{n}A\}$. Sabiendo esto, podemos calcular las trazas (\ref{tr1}) y (\ref{tr2}).

\begin{equation}
\label{tr1solved}
Tr_{1}(\lambda_{m})=\xi_{0}^{2}\braket{b_{0}b_{0}^{\dag}}_{th}\left[\left(\frac{A_{-}}{A_{+}}\right)\delta_{\lambda_{m},i\nu(t)}+\delta_{\lambda_{m},-i\nu(t)}\right]
\end{equation}

\begin{equation}
\label{tr2solved}
Tr_{2}(\lambda_{m})=\xi_{0}^{2}\braket{b_{0}b_{0}^{\dag}}_{th}\left[\delta_{\lambda_{m},i\nu(t)}+\left(\frac{A_{-}}{A_{+}}\right)\delta_{\lambda_{m},-i\nu(t)}\right]
\end{equation}

donde $\braket{b_{0}b_{0}^{\dag}}_{th}=Tr\{b_{0}b_{0}^{\dag}\rho_{st}\}$, y por tanto, contiene la información sobre la taza de enfriamiento promedio y la escala de tiempo, en la cual, el estado estacionario del oscilador mecánico es alcanzada. Volviendo a (\ref{separacion}), los elementos que contienen la información sobre la cavidad, es decir, la parte operacional de la ecuación es

\begin{equation}
\label{operadores1}
\mathcal{P}_{c}^{\lambda_{c}=0}\left[F,\mathcal{P}_{c}^{\lambda_{c}}(\psi F)\right]
\end{equation}

\begin{equation}
\label{operadores2}
\mathcal{P}_{c}^{\lambda_{c}=0}\left[F,\mathcal{P}_{c}^{\lambda_{c}}(F\psi)\right]
\end{equation}

donde las proyecciones están hechas con la base de disipación  del atenuador [RefStenholm][RefEnglert], recuerde que $Tr\left\{\check{\rho}^{\lambda}\hat{\rho}^{\lambda^{\prime}}\right\}=\delta_{\lambda,\lambda^{\prime}}$




\begin{equation}
\hat{\rho}_{n}^{(k)}=
											\begin{cases}
													a^{\dag k}\left(-1\right)^{a^{\dag}a+n}\binom{n+k}{a^{\dag}a+k}     & \quad k\geq0\\
													\left(-1\right)^{a^{\dag}a+n}\binom{n+|k|}{a^{\dag}a+|k|}a^{|k|}  & \quad k<0\\
												\end{cases}
\end{equation}

\begin{equation}
\check{\rho}_{n}^{(k)}=
\begin{cases}
												\frac{a^{n}a^{\dag n+k}}{(n+k)!}     & \quad k\geq0\\
												\frac{a^{n+|k|}a^{\dag n}}{(n+|k|)!} & \quad k<0\\
												\end{cases}
\end{equation}


con eigenvalores $\lambda_{c}(k,n)=ik\Delta_{L}-(2n+|k|)\kappa$ para enteros $k=(-\infty,\infty)$ y $n=(0,\infty)$. Tomando (\ref{operadores1}) para desarrollarlo, obtenemos


\begin{equation}
\label{solvingoperadores1}
\mathcal{P}_{c}^{\lambda_{c}=0}\left[F,\mathcal{P}_{c}^{\lambda_{c}}(\psi F)\right]=\hat{\rho}_{0}^{(0)}Tr\left[F,\hat{\rho}_{n}^{(k)}\right]Tr\left\{\check{\rho}_{n}^{(k)}\psi F\right\}
\end{equation}

debe notarse que el eigenelento dual $\check{\rho}_{0}^{(0)}$ actua como el operador unidad. La primera parte de (\ref{solvingoperadores1}) queda como

\begin{equation}
Tr\left[F,\hat{\rho}_{n}^{(k)}\right]=-\hbar g \left(    \alpha_{0}\left[a^{\dag},\hat{\rho}_{n}^{(k)}\right]   +  \alpha_{0}^{*}\left[a,\hat{\rho}_{n}^{(k)}\right]     \right)
\end{equation}

puesto que $[a^{\dag}a,\hat{\rho}_{n}^{(k)}]=0$. Resolviendo por separado los conmutadores para los diferentes casos de valores de $k$

\begin{equation}
Tr\left[a^{\dag},\hat{\rho}_{n}^{(k\geq0)}\right]=0
\end{equation}

\begin{eqnarray}
\label{conmutadorcreacion}
Tr\left[a^{\dag},\hat{\rho}_{n}^{(k<0)}\right]& = & Tr\left[a^{\dag},f_{n}^{|k|}(a^{\dag},a)a^{|k|}\right] \nonumber \\
                                          & = & \sum_{m}\left[m f_{n}^{|k|}(m-1)-(m+1)f_{n}^{|k|}(m)\right]\delta_{k,-1} \nonumber \\
																					%& = & -2\delta_{\lambda_{c},-i\Delta_{L}-\kappa}
\end{eqnarray}

para el cual se ha definido $f_{n}^{(k)}(a^{\dag}a)=(-1)^{a^{\dag}a + n}\binom{n+(k)}{a^{\dag}a +(k)}$. Y de igual manera para el conmutador dual%y la sumatoria dentro de (\ref{conmutadorcreacion}) se anula para todo $n>0$.


\begin{equation}
Tr\left[a,\hat{\rho}_{n}^{(k<0)}\right]=0
\end{equation}


\begin{eqnarray}
\label{conmutadoraniquilacion}
Tr\left[a^{\dag},\hat{\rho}_{n}^{(k<0)}\right]& = & Tr\left[a^{\dag},a^{k}f_{n}^{k}(a^{\dag},a)\right] \nonumber \\
                                          & = & \sum_{m}\left[(m+1)f_{n}^{k}(m)-m f_{n}^{k}(m-1)\right]\delta_{k,-1} \nonumber \\
																					%& = & 2\delta_{\lambda_{c},i\Delta_{L}-\kappa}
\end{eqnarray}

Analicemos la serie que dió como resultado de los conmutadores, note que es la misma en las ecuaciones (\ref{conmutadorcreacion}) y (\ref{conmutadoraniquilacion})

\begin{equation}
\label{serie}
g(n)=\sum_{m=0}^{\infty}\left[(m+1)f_{n}^{(1)}(m)-m f_{n}^{(1)}(m-1)\right]
\end{equation}

En la Figura (\ref{fig:gn}) se muestran los elementos de la serie término a término para varios $n$. Note que independientemente de que valor sea asignado a $n$, la serie siempre se corta cuando el número de términos $m=n$ y es igual a cero debido a su simetría con respecto del eje horizontal

Con lo cual, se concluye, que para el subespacio seleccionado por la proyección $\mathcal{P}=\mathcal{P}_{c}^{\lambda_{c}=0}\mathcal{P}_{m}^{\lambda_{m}=0}$ la cavidad, no evoluciona. La no evolución de la cavidad en tal proyección, es debido, a que para las aproximaciones hechas, la cavidad decae más rápidamente que el oscilador mecánico, teniendo como resultado un operador densidad

\begin{equation}
\mathcal{P}\rho=\ket{0}\bra{0}\otimes \rho_{st}
\end{equation}

recuperando lo mencionado en [RefPablo]. Sin embargo, este resultado no nos resuelve muchas dudas, y por tanto debemos de buscar otras proyecciones que nos proporcionen más información, proyecciones que apesar de que decaigan, no lo hagan tan rápidamente. Para elegir los subespacios en los que nos centraremos, tenemos que fijarnos en el eigenvalor que los representa, debido a que este mismo eigenvalor dictará que tan rápido se esfuman de la evolución, en especial, en el dado por la descomposición espectral de la cavidad $\lambda_{c}=ik\Delta_{L}-(2n+|k|)\kappa$. La parte real nos dará un decaimiento exponencial, y la parte imaginaria nos dará una oscilación alrededor del decaimiento.

Por tanto, la selección del subespacio está dado por la parte real, y es en la que nos centraremos. Es importante notar que al igual que en la ecuación (\ref{solvingoperadores1}), el eigenelemento $\hat{\rho}_{0}^{0}=\ket{0}\bra{0}$ nos decía que información estaba a nuestro alcance, debido a que en todo el desarrollo era el único elemento totalmente con caracter de operador, en este caso sólo la evolución del estado de ocupación de menor energía $\psi_{00}$, en las siguientes proyecciones a realizar, los elementos $\hat{\rho}_{n}^{k}$ tendrán un rol similar.

Sí consideramos que las proyecciones que hemos elegido generan un estado separable de la misma manera que (\ref{estadoseparable}), la ecuación (\ref{evolucion}) se convierte en

\begin{equation}
\label{evolucion}
\dot{\psi}=\sum_{\lambda\neq\left\{\lambda^{\prime}\right\}}\sum_{l}\frac{K_{l}^{\lambda_{m}}(t)}{\lambda_{c}+\lambda_{m}+il\omega}Tr_{m}\{\mathcal{P}\mathcal{L}_{1}\mathcal{P}_0^{\lambda}\mathcal{L}_{1}\rho_{st}\psi\}
\end{equation}

donde el conjunto $\left\{\lambda^{\prime}\right\}$ contendrá los eigenvalores que privilegiaremos. Los subespacios que elegiremos, en orden de importancia, son aquellos en los que $\lambda_{c}=-1,-2$. Es necesario resaltar que puesto que hay dos proyecciones con el mismo valor real $\lambda_{c}=-1$ y tres para $\lambda_{c}=-2$, tendremos que trabajar en el esquema de interacción (ignoraremos la tipificación de las variables que tendrían que ir primadas por el cambio de esquema), con la finalidad de quitarnos los problemas de la parte imaginaria del eigenvalor de la cavidad, es decir,

\begin{equation}
\mathcal{L}_{0}\rho=\mathcal{L}_{\kappa}\rho+\frac{1}{i\hbar}\left[H_{mec}(t),\rho\right]
\end{equation}

con lo cual hemos quitado el término $\left[H_{cav},\rho\right]$ lo cual contenía los términos asociados al desfase $\Delta_{L}$. Con este problema arreglado las sumas de proyectores

\begin{equation}
\mathcal{P}_{c}^{(\lambda_{c}=-\kappa)}=\mathcal{P}_{n=0}^{k=1}+\mathcal{P}_{n=0}^{k=-1}
\end{equation}

\begin{equation}
\mathcal{P}_{c}^{(\lambda_{c}=-2\kappa)}=\mathcal{P}_{n=1}^{k=0}+\mathcal{P}_{n=0}^{k=2}+\mathcal{P}_{n=0}^{k=-2}
\end{equation}

también serán proyectores y generará el subespacio más general para cada eigenvalor. Es decir, generaremos proyecciones en las cuales el sistema es separable, el oscilador mecánico está en su estado estacionario y el subespacio de la cavidad, decae como $e^{\lambda_{c}t}$. Note que las ecuaciones (\ref{separacion}), (\ref{tr1solved}) y (\ref{tr2solved}) quedan intactas, y nos tendremos que concentrar sólo en las proyecciones y grados de libertad de la cavidad.

Iniciando por $\mathcal{P}_{c}^{(\lambda_{c}=-\kappa)}$

\begin{equation}
\mathcal{P}_{0}^{1}\left[F,\mathcal{P}^{(k)}_{n}(A)\right]=-\alpha_{0}\ket{1}\bra{0}Tr{\check{\rho}_{(k)}^{n}A}\delta_{\lambda_{c},0}
\end{equation}

\begin{equation}
\mathcal{P}_{0}^{-1}\left[F,\mathcal{P}^{(k)}_{n}(A)\right]=\alpha_{0}^{*}\ket{0}\bra{1}Tr{\check{\rho}_{(k)}^{n}A}\delta_{\lambda_{c},0}
\end{equation}

donde por comparación con (\ref{operadores1}) y (\ref{operadores2}) $A=\psi F$ ó $A=F\psi$. Y entonces (\ref{evolucion}) quedaría

\begin{align}
\dot{\psi}=\xi_{0}^2\braket{b_{0}b_{0}^{\dag}}_{th}\sum_{l}&\left[\frac{K_{l}^{i\nu(t)}}{i\nu(t)+il\omega}\left(\frac{A_{-}}{A_{+}}\right)Tr\left\{\psi F\right\} + \frac{K_{l}^{-i\nu(t)}}{-i\nu(t)+il\omega}Tr\left\{\psi F\right\}\right. \nonumber \\
-&\left. \frac{K_{l}^{i\nu(t)}}{i\nu(t)+il\omega}Tr\left\{F\psi\right\}-\frac{K_{l}^{-i\nu(t)}}{-i\nu(t)+il\omega}\left(\frac{A_{-}}{A_{+}}\right)Tr\left\{F\psi\right\}\right]\hat{\sigma}_{01}\nonumber
\end{align}

con

\begin{equation*}
\hat{\sigma}_{01}=\left(\alpha_{0}^{*}\ket{0}\bra{1}-\alpha_{0}\ket{1}\bra{0}\right) 
\end{equation*}

Y puede llegar a simplificarse, debido a tratamiento de las sumatorias (y la promediación de los términos oscilantes), a

\begin{equation}
\label{psi01}
\dot{\psi}_{01}=2\xi_{0}^2\braket{b_{0}b_{0}^{\dag}}\alpha_{0}^{*}(1-\frac{A_{-}}{A_{+}})\sum_{l}\frac{i\braket{K^{i\nu(t)}_{l}}}{\nu_{0}+l\omega}(\alpha_{0}\psi_{01}+\alpha_{0}^{*}\psi_{10})
\end{equation}

donde se ha tomado en cuenta que en el subespacio donde estamos trabajando sólo tenemos al alcance información sobre los elementos $\psi_{01}$, $\psi_{10}$. Y por comprobación, también podemos obtener

\begin{equation}
\dot{\psi}_{10}=-2\xi_{0}^2\braket{b_{0}b_{0}^{\dag}}\alpha_{0}(1-\frac{A_{-}}{A_{+}})\sum_{l}\frac{i\braket{K^{i\nu(t)}_{l}}}{\nu_{0}+l\omega}(\alpha_{0}\psi_{01}+\alpha_{0}^{*}\psi_{10})
\end{equation}

note que en lugar de haber proyectado la ecuación de evolución de $\psi$ en $\bra{1}\dot{\psi}\ket{0}$ pudimos haber simplemente sacado el complejo conjugado de \ref{psi01}.
Aún mas importante será la proyección $\mathcal{P}_{c}^{(\lambda_{c}=-2\kappa)}$, debido a que por la forma de su elemento $\mathcal{P}_{n=1}^{k=0}$ podremos obtener información sobre los elementos $\psi_{00}$ y $\psi_{11}$
